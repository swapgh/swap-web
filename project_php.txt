    private Map<Integer, BufferedImage> tiles;

    public TileManager(int tileSize) {

    private void loadTiles() {

            // Cargar imÃ¡genes para cada tipo de terreno

            tiles.put(1, ImageIO.read(getClass().getResource("/tiles/water.png")));  // Agua

        } catch (IOException e) {

            // En caso de error, usamos colores como fallback

        try (BufferedReader reader = new BufferedReader(new FileReader(filePath))) {
            String line;

            // Leer todas las lÃ­neas del archivo

            int lineIndex = 0;
            
            // Leer tamaÃ±o del mapa
            String[] sizeParts = lines.get(lineIndex++).split(" ");

                throw new MapFormatException("Formato incorrecto para el tamaÃ±o del mapa");
            }
            
            int width = Integer.parseInt(sizeParts[0]);
            int height = Integer.parseInt(sizeParts[1]);
            
            // Leer datos del mapa
            int[][] map = new int[height][width];
            for (int y = 0; y < height; y++) {
                if (lineIndex >= lines.size()) {
                    throw new MapFormatException("No hay suficientes filas en el mapa");
                }
                
                String[] rowParts = lines.get(lineIndex++).split(" ");
                if (rowParts.length < width) {
                    throw new MapFormatException("Fila " + y + " incompleta en el mapa");
                }
                
                for (int x = 0; x < width; x++) {
                    map[y][x] = Integer.parseInt(rowParts[x]);
                }
            }
            
            // Leer posiciÃ³n inicial del personaje (si existe)
            int playerX = 0;
            int playerY = 0;
            if (lineIndex < lines.size()) {
                String[] playerParts = lines.get(lineIndex).split(" ");
                if (playerParts.length >= 2) {
                    playerX = Integer.parseInt(playerParts[0]);
                    playerY = Integer.parseInt(playerParts[1]);
                }
            }
